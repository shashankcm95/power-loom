## Empirical pre-validation findings

**3 static spawnSync sites in auto-store-enrichment.js bumpSelfImproveCounters (lines 236-265)**:

| Line | Site | Frequency | Latency cost (per call) |
|------|------|-----------|-------------------------|
| 243 | `bump-turn` | 1× per Stop | ~50-100ms |
| 250 | `bump --signal` (loop body) | up to 20× per Stop (`signals.slice(0, 20)`) | ~50-100ms each |

Worst-case: 22 sequential subprocess invocations × ~50-100ms = **1.1-2.2 seconds latency in user-perceptible Stop-event window** when 20 unique signals are detected AND the scan-cadence trigger fires.

**self-improve-store.js current state**:
- NO `module.exports` (purely CLI-driven; top-level switch dispatch at lines 425-447)
- Refactor requires: (a) wrap CLI dispatch in `require.main === module` guard; (b) add `module.exports`; (c) author new `bumpBatch(signals)` programmatic-surface function
- Existing CLI commands (`cmdBump`, `cmdBumpTurn`, `cmdScan`) preserved as-is for backward compatibility (other consumers may invoke via CLI)

**No direct test surface for bumpSelfImproveCounters**: observable behavior is via filesystem state changes (counters file, pending queue). Refactor preserves filesystem semantics by design.

## Implementation strategy

### Sub-phase 1 — Add programmatic surface to self-improve-store.js

1. Wrap CLI dispatch in `if (require.main === module)` guard at line 425
2. Add `module.exports = { bumpBatch, cmdBump, cmdBumpTurn, cmdScan, cmdPending, cmdDismiss, cmdPromote, cmdReset, cmdStats }`
3. Add new `bumpBatch(signals)` function:

```js
/**
 * Batched in-process call: bump turn counter + bump each signal (up to 20)
 * + conditional scan. Replaces 22-spawnSync worst-case from
 * auto-store-enrichment.js bumpSelfImproveCounters at HT.1.14.
 *
 * @param {string[]} signals - array of signals (each "type:value" form); up to 20 processed
 * @returns {{shouldScan: boolean, signalsBumped: number, scanResult: object|null, turnCounter: number}}
 */
function bumpBatch(signals) {
  let shouldScan = false;
  let signalsBumped = 0;
  let scanResult = null;
  let turnCounter = 0;
  const sigs = Array.isArray(signals) ? signals.slice(0, 20) : [];
  
  // Single withLock for counter mutations (better atomicity than 22 separate
  // lock acquisitions; matches in-process state mutation pattern).
  withLock(COUNTERS_PATH + '.lock', () => {
    const counters = loadCounters();
    const now = new Date().toISOString();
    
    // bump-turn
    counters.turnCounter += 1;
    turnCounter = counters.turnCounter;
    if (counters.turnCounter - counters.lastScanTurn >= SCAN_TURN_INTERVAL) {
      shouldScan = true;
    }
    
    // bump each signal
    for (const signal of sigs) {
      const entry = counters.signals[signal] || { count: 0, firstSeen: now, lastSeen: now };
      entry.count += 1;
      entry.lastSeen = now;
      counters.signals[signal] = entry;
      signalsBumped++;
    }
    
    writeAtomic(COUNTERS_PATH, counters);
  });
  
  // Conditional scan (separate lock acquisition to match cmdScan's nested withLock pattern)
  if (shouldScan) {
    // Inline scan logic — call cmdScan would write to console.log; keep silent for in-process call
    let result;
    withLock(COUNTERS_PATH + '.lock', () => {
      withLock(PENDING_PATH + '.lock', () => {
        const counters = loadCounters();
        const pending = loadPending();
        const knownSignatures = new Set(pending.candidates.map((c) => c.signal));
        const newCandidates = [];
        const autoGraduated = [];
        for (const [signal, entry] of Object.entries(counters.signals)) {
          if (knownSignatures.has(signal)) continue;
          if (entry.count < THRESHOLDS.candidate) continue;
          const kind = inferKindFromSignal(signal);
          const risk = KIND_RISK[kind] || 'medium';
          const candidate = {
            id: newCandidateId(), kind, signal,
            occurrences: entry.count,
            firstSeen: entry.firstSeen,
            lastSeen: entry.lastSeen,
            risk,
            summary: signalToSummary(signal, entry),
            proposedAction: signalToProposedAction(signal, kind),
            status: 'pending',
            createdAt: new Date().toISOString(),
          };
          if (risk === 'low' && entry.count >= THRESHOLDS.autoGraduate) {
            candidate.status = 'auto-graduated';
            executeGraduation(candidate);
            autoGraduated.push(candidate);
          } else {
            newCandidates.push(candidate);
          }
        }
        pending.candidates.push(...newCandidates, ...autoGraduated);
        counters.lastScanAt = new Date().toISOString();
        counters.lastScanTurn = counters.turnCounter;
        writeAtomic(COUNTERS_PATH, counters);
        writeAtomic(PENDING_PATH, pending);
        result = {
          newCandidates: newCandidates.length,
          autoGraduated: autoGraduated.length,
          totalPending: pending.candidates.filter((c) => c.status === 'pending').length,
        };
      });
    });
    scanResult = result;
  }
  
  return { shouldScan, signalsBumped, scanResult, turnCounter };
}
```

Note on scan body duplication: extracting a shared `_scanInternal()` would be cleaner but requires refactoring `cmdScan` (existing CLI surface). HT.1.14 scope keeps the duplication for mechanical-refactor discipline; HT.2 sweep candidate to extract shared internal helper.

### Sub-phase 2 — Refactor auto-store-enrichment.js bumpSelfImproveCounters

Replace lines 236-265 with in-process call:

```js
function bumpSelfImproveCounters(signals) {
  if (!SELF_IMPROVE_SCRIPT) return null;
  // HT.1.14: batched in-process call (was 3 spawnSync sites worst-case 22 calls).
  // require() the module directly; cached after first load. ADR-0001 fail-soft
  // invariant 2 preserved via try/catch + log on error.
  try {
    const store = require(SELF_IMPROVE_SCRIPT);
    const result = store.bumpBatch(signals);
    return { shouldScan: result.shouldScan, signalsBumped: result.signalsBumped };
  } catch (err) {
    log('self_improve_failed', { error: err.message });
    return null;
  }
}
```

The `require(SELF_IMPROVE_SCRIPT)` resolves to the path returned by `resolveSelfImproveScript()` — works in both deployed-marketplace install (`~/.claude/scripts/`) and local-checkout (`~/Documents/claude-toolkit/scripts/`). Module load is one-time-cached by Node's require cache; subsequent Stop events reuse the loaded module.
